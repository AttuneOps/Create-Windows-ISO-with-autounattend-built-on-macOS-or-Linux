if sestatus | grep -i "SELinux status:" | grep -q "enabled"
then
    line_count=$(getsebool -a | grep samba_export | wc -l)
    
    if [ $line_count -eq 2 ]
    then
        echo "ok"
    else
        echo "wrong number of occurrences for samba_export"
        exit 1
    fi
else
    echo "SElinux is disabled"
fi
